üìù Component Code Examples

VentryButton.jsx
import { Button } from 'flowbite-react';
import { ventryTheme } from './VentryTheme';

export function VentryButton({ 
  children, 
  variant = 'primary', 
  size = 'md', 
  loading = false,
  ...props 
}) {
  const getVariant = () => {
    switch(variant) {
      case 'secondary': return 'ventrySecondary';
      case 'outline': return 'ventryOutline';
      default: return 'ventry';
    }
  };

  return (
    <Button
      theme={ventryTheme.button}
      color={getVariant()}
      size={size}
      disabled={loading}
      className="transition-all duration-200 hover:-translate-y-0.5"
      {...props}
    >
      {loading ? (
        <>
          <svg className="animate-spin -ml-1 mr-3 h-5 w-5 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
            <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
            <path className="opacity-75" fill="currentColor" d="m4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
          </svg>
          Loading...
        </>
      ) : children}
    </Button>
  );
}
